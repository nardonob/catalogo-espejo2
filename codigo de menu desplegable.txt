<script>
    function toggleDropdown(button) {
        // Cerrar todos los otros dropdowns
        const allDropdowns = document.querySelectorAll('.dropdown-content');
        const allButtons = document.querySelectorAll('.category-btn');
        const currentDropdown = button.nextElementSibling;
        
        allDropdowns.forEach(dropdown => {
            if (dropdown !== currentDropdown) {
                dropdown.classList.remove('show');
            }
        });
        
        allButtons.forEach(btn => {
            if (btn !== button && !btn.classList.contains('all-products-btn')) {
                btn.classList.remove('active');
            }
        });
        
        // Toggle del dropdown actual
        currentDropdown.classList.toggle('show');
        button.classList.toggle('active');
    }

    // Cerrar dropdowns al hacer clic fuera
    window.onclick = function(event) {
        if (!event.target.matches('.category-btn') && !event.target.matches('.arrow')) {
            const dropdowns = document.querySelectorAll('.dropdown-content');
            const buttons = document.querySelectorAll('.category-btn:not(.all-products-btn)');
            
            dropdowns.forEach(dropdown => {
                dropdown.classList.remove('show');
            });
            
            buttons.forEach(button => {
                button.classList.remove('active');
            });
        }
    }
</script>
